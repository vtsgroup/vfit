## 01) Tech Stack Definitivo

```yaml
FRONTEND:
  Framework: Next.js 15.2+ (App Router, Static Export)
  Runtime: Node.js 20 LTS
  Language: TypeScript 5.4+ (strict mode)
  Deploy: Cloudflare Pages (via Wrangler CLI)

UI & STYLING:
  CSS: TailwindCSS 4.0
  Components: shadcn/ui (New York variant)
  Animations: Framer Motion 11+
  Icons: Lucide React
  Fonts: Inter Variable (heading) + DM Sans (body) via next/font

STATE & DATA:
  Client State: Zustand 5
  Server/Async: TanStack Query v5 (React Query)
  Forms: React Hook Form + Zod validation
  URL State: nuqs (type-safe search params)

PWA:
  Plugin: @ducanh2912/next-pwa (substituto moderno do next-pwa)
  Cache: Workbox (runtime caching de assets)
  Manifest: /public/manifest.json
  Service Worker: Auto-generated + custom offline page

SEO TOOLING:
  Metadata: Next.js Metadata API (generateMetadata)
  Schemas: next-seo + custom JSON-LD components
  Sitemap: next-sitemap (auto-generated on build)
  Analytics: GA4 + GTM + Plausible (privacy-first backup)
  A/B Testing: Custom hook + Cloudflare Workers (edge)

QUALIDADE:
  Lint: ESLint 9 flat config + eslint-plugin-jsx-a11y
  Format: Prettier + @trivago/prettier-plugin-sort-imports
  Types: tsc --noEmit on CI
  Tests: Vitest + Testing Library + Playwright (E2E)
  Bundle: @next/bundle-analyzer (keep < 150kb first load)
```

```typescript
import type { NextConfig } from 'next';

const nextConfig: NextConfig = {
  output: 'export',               // Static export for Cloudflare Pages
  trailingSlash: false,           // Clean URLs (no trailing /)
  images: {
    unoptimized: true,            // Required for static export
    formats: ['image/avif', 'image/webp'],
  },
  experimental: {
    mdxRs: true,                  // Rust-based MDX compiler
  },
  headers: async () => [
    {
      source: '/(.*)',
      headers: [
        { key: 'X-Content-Type-Options', value: 'nosniff' },
        { key: 'X-Frame-Options', value: 'DENY' },
        { key: 'X-XSS-Protection', value: '1; mode=block' },
        { key: 'Referrer-Policy', value: 'strict-origin-when-cross-origin' },

    },
  ],
  redirects: async () => [
    // Legacy URL redirects
    { source: '/blog/melhor-app-personal-trainer', destination: '/blog/app-para-personal-trainer', permanent: true },
  ],
};

export default nextConfig;
```

## 05) Core Web Vitals Targets

| Métrica | Target | Como garantir |
|---------|--------|---------------|
| **LCP** | < 2.0s | Hero com next/image priority, font preload, no render-blocking CSS |
| **FID/INP** | < 100ms | Hydration mínima, defer non-critical JS, use `startTransition` |
| **CLS** | < 0.05 | Explicit width/height em todas imagens, font-display: swap, skeleton loaders |
| **TTFB** | < 200ms | Cloudflare CDN edge cache, static export |
| **FCP** | < 1.2s | Critical CSS inline, font preload, minimal above-fold JS |
| **Bundle** | < 150kb | Tree-shaking, dynamic imports, analyze with @next/bundle-analyzer |

```tsx
  children,
  className,
  variants = fadeInUp,
  once = true,
}: {
  children: React.ReactNode;
  className?: string;
  variants?: typeof fadeInUp;
  once?: boolean;
}) {
  const ref = useRef(null);
  const isInView = useInView(ref, { once, margin: '-100px 0px' });

  return (

  );

// 1. NUNCA animar layout properties (width, height, top, left)
// 2. APENAS animar: opacity, transform (translate, scale, rotate)
// 3. Sempre usar will-change: transform em elementos animados
// 4. prefers-reduced-motion: desligar animações
```

## 26) A/B Testing — Sistema Customizado (Edge)

### Arquitetura

```
Visitante → Cloudflare Worker (edge) → atribui variante → cookie 30d
→ Next.js lê cookie → renderiza variante → GA4 event com variant ID
```

### Hook useABTest

```tsx
// src/hooks/useABTest.ts
'use client';

import { useEffect, useState } from 'react';
import { trackEvent } from '@/lib/analytics';

interface ABTestConfig {
  testId: string;
  variants: string[];
  weights?: number[]; // [0.5, 0.5] = 50/50
}

export function useABTest({ testId, variants, weights }: ABTestConfig) {
  const [variant, setVariant] = useState<string | null>(null);

  useEffect(() => {
    // Check existing assignment
    const stored = document.cookie
      .split('; ')
      .find(row => row.startsWith(`ab_${testId}=`))
      ?.split('=')[1];

    if (stored && variants.includes(stored)) {
      setVariant(stored);
      return;
    }

    // Assign new variant
    const rand = Math.random();
    const w = weights || variants.map(() => 1 / variants.length);
    let cumulative = 0;
    let assigned = variants[0];
    for (let i = 0; i < variants.length; i++) {
      cumulative += w[i];
      if (rand <= cumulative) { assigned = variants[i]; break; }
    }

    // Set cookie for 30 days
    document.cookie = `ab_${testId}=${assigned}; max-age=${30*86400}; path=/; SameSite=Lax`;
    setVariant(assigned);

    // Track impression
    trackEvent('ab_test_impression', { test_id: testId, variant: assigned });
  }, [testId, variants, weights]);

  return variant;
}
```

```tsx
  email: z.string().email('Email inválido'),
  name: z.string().min(2, 'Nome é obrigatório'),
});

type FormData = z.infer<typeof schema>;

export function LeadMagnetForm({
  magnetName,
  headline,
  description,
  buttonText = 'Baixar grátis',
}: {
  magnetName: string;
  headline: string;
  description: string;
  buttonText?: string;
}) {
  const [submitted, setSubmitted] = useState(false);
  const { register, handleSubmit, formState: { errors } } = useForm<FormData>({
    resolver: zodResolver(schema),
  });

  const onSubmit = async (data: FormData) => {
    trackEvent('lead_magnet_submit', {
      magnet_name: magnetName,
      email_domain: data.email.split('@')[1],
    });
    // TODO: enviar para API de email (Resend, etc.)
    setSubmitted(true);
  };

  return (
    <div className="bg-surface-100 border border-brand-500/20 rounded-2xl p-6 shadow-glow-sm">
      <AnimatePresence mode="wait">
        {!submitted ? (
          <motion.div key="form" initial={{ opacity: 1 }} exit={{ opacity: 0 }}>
            <h3 className="text-display-sm text-surface-900 mb-2">{headline}</h3>
            <p className="text-body-md text-surface-600 mb-4">{description}</p>
            <div
              className="space-y-3"
              onSubmit={handleSubmit(onSubmit)}
            >
              <input
                {...register('name')}
                placeholder="Seu nome"
                className="w-full px-4 py-3 bg-surface-200 border border-surface-300 rounded-xl text-surface-800 placeholder:text-surface-500 focus:border-brand-500 focus:ring-1 focus:ring-brand-500 transition"
              />
              {errors.name && <p className="text-error text-body-sm">{errors.name.message}</p>}
              <input
                {...register('email')}
                type="email"
                placeholder="seu@email.com"
                className="w-full px-4 py-3 bg-surface-200 border border-surface-300 rounded-xl text-surface-800 placeholder:text-surface-500 focus:border-brand-500 focus:ring-1 focus:ring-brand-500 transition"
              />
              {errors.email && <p className="text-error text-body-sm">{errors.email.message}</p>}
              <button
                type="button"
                onClick={handleSubmit(onSubmit)}
                className="w-full bg-gradient-cta text-white font-semibold py-3 rounded-xl shadow-glow-sm hover:shadow-glow-md transition-all"
              >
                {buttonText}
              </button>
              <p className="text-body-xs text-surface-500 text-center">
                Zero spam. Cancele quando quiser.
              </p>
            </div>
          </motion.div>
        ) : (
          <motion.div key="success" initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }}>
            <div className="text-center py-4">
              <span className="text-4xl mb-4 block">🎉</span>
              <h3 className="text-display-sm text-surface-900 mb-2">Pronto!</h3>
              <p className="text-body-md text-surface-600">Verifique seu email para acessar o material.</p>
            </div>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}
```
